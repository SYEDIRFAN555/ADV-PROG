#include<stdio.h>
int main()
{
	int a[10][10],b[10][10],i,j,x,y;
	scanf("%d%d",&x,&y);
	printf("Enter 1st Matrix\n");
	for(i=0;i<x;i++)
	{
	for(j=0;j<y;j++)
	scanf("%d",&a[i][j]);
	}
	printf("Enter 2nd Matrix\n");
	for(i=0;i<x;i++)
	{
		for(j=0;j<y;j++)
		scanf("%d",&b[i][j]);
	}
	printf("Resultant Matrix\n");
	for(i=0;i<x;i++)
	{
		for(j=0;j<y;j++)
		printf("%d ",a[i][j]+b[i][j]);
		printf("\n");
	}
}
