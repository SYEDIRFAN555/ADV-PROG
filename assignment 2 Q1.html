#include <stdio.h>  
#define M 3 
#define N 3   
void form(int aa[], int ma[M][N]) 
{ 
    int top = 0,i,bottom = M - 1,left = 0,right = N - 1,index = 0; 
    while (1) { 
        if (left > right) 
            break; 
        for (i = left; i <= right; i++) 
            ma[top][i] = aa[index++]; 
        top++; 
        if (top > bottom) 
            break; 
        for (i = top; i <= bottom; i++) 
            ma[i][right] = aa[index++]; 
        right--; 
        if (left > right) 
            break; 
        for (i = right; i >= left; i--) 
            ma[bottom][i] = aa[index++]; 
        bottom--; 
        if (top > bottom) 
            break; 
        for (i = bottom; i >= top; i--) 
            ma[i][left] = aa[index++]; 
        left++; 
    } 
} 
void print(int ma[M][N]) 
{ 
   int i,j;
    for (i = 0; i < M; i++) { 
        for (j = 0; j < N; j++)
        printf("%d ",ma[i][j]);
        printf("\n");
    } 
} 
int main() 
{ 
    int aa[] 
        = { 1, 2, 3, 4, 5, 6, 
            7, 8, 9}; 
    int ma[M][N]; 
    form(aa, ma); 
    print(ma); 
    return 0; 
}
