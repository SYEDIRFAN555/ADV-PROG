#include<stdio.h>
#include<string.h>
void main()
{
	char str[100];
	int i,length;
	int flag=0;
	scanf("%s",str);
	length=strlen(str);
	for(i=0;i<length;i++)
	{
		if(str[i]!=str[length-i-1])
		{
		 flag++;
		 break;
		}
	}
	if(flag==1)
	printf("%s is not a palindrome",str);
	else
	printf("%s is a palindrome",str);
}
