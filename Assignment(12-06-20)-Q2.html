#include<stdio.h>
int main()
{
	int a[10][10],b[10][10],i,j,x1,y1,x2,y2,f=1;
	scanf("%d%d",&x1,&y1);
	scanf("%d%d",&x2,&y2);
	if(x1==x2 && y1==y2)
	{
		printf("Enter 1st Matrix\n");
		for(i=0;i<x1;i++)
		{
			for(j=0;j<y1;j++)
			scanf("%d",&a[i][j]);
		}
		printf("Enter 2nd Matrix\n");
		for(i=0;i<x2;i++)
		{
			for(j=0;j<y2;j++)
			scanf("%d",&b[i][j]);
		}
		for(i=0;i<x1;i++)
		{
			for(j=0;j<y1;j++)
			{
				if(a[i][j]!=b[i][j])
				{
					f=0;
					break;
				}
			}
		}
		if(f==0)
		printf("given matrices are not equal\n");
		else
		printf("given matrices are equal\n");
	}
	else
	printf("matrices cannot be equal with different order\n");
}
