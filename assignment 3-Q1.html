#include<stdio.h> 
int main()
{    
    int n,p,m,i=0,a,b;
    scanf("%d%d%d",&p,&m,&n);
    while(n!=0)
    {    
    a=i%3,b=i/3; 
    switch(a) 
    {
    case 0:if(i==0)
    printf("%d ",p);
    else
    printf("%d ",(b+1)*p+b*m);
    break; 
    case 1:if(i==1)
    printf("%d ",i*m);
    else
    printf("%d ",b*p+(b+1)*m);
    break;
    case 2:printf("%d ",(b+1)*p+(b+1)*m);
    break;
    }
        n--;
        i++;
    }
    return 0;
}
